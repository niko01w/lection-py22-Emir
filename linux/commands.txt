как открыть термирал =  ctrl=alt=l 
 символ home directop = ~
  символ корневой директории (папки) = / 
   обозначает текущую директорию = .
   обозначает прошлую директорию = ..

                                           термирал
     ls = все директории в текущей директории
      cd = переход из директории в директорию
       mkdir = создать директорию
	touch = создать файл (txt. docs.py)
         pwd = показывает путь директории  
	  rmdir = (remove directory) удаляет указанную директорию
	   rm = удаляет файлы
            rm - rf = принудительно удаляет все 
 	     mv = перемещяет либо переименовывает файл либо директорию, в зависимости от указанного второго аргумента 
	      cat = выводит содержимое файла
